function combine(...some) {
	const sum = {};
	some.forEach((obj) => {
		Object.keys(obj).forEach((key) => {
			if (!sum[key]) { sum[key] = obj[key] }
			else { sum[key] = sum[key] + obj[key] }
		})
	})
	return sum
}
